const express = require('express');
const cors = require('cors');
const ytdl = require('ytdl-core');

const app = express();
app.use(cors());
app.use(express.json());

app.post('/api/fetch', async (req, res) => {
  const { url } = req.body;
  if (!ytdl.validateURL(url)) {
    return res.json({ error: 'Invalid YouTube URL' });
  }
  try {
    const info = await ytdl.getInfo(url);
    const videoDetails = info.videoDetails;
    
    const formats = info.formats
      .filter(f => f.hasAudio && f.hasVideo && f.container)
      .map(f => ({
        url: f.url,
        ext: f.container,
        resolution: f.qualityLabel || 'Audio Only',
        filesize: f.contentLength
          ? (Number(f.contentLength) / (1024 * 1024)).toFixed(2) + ' MB'
          : 'Unknown'
      }));

    res.json({
      title: videoDetails.title,
      thumbnail: videoDetails.thumbnails[videoDetails.thumbnails.length - 1].url,
      channel: videoDetails.author.name,
      formats
    });
  } catch (error) {
    console.error(error);
    res.json({ error: 'Failed to fetch video info' });
  }
});

const PORT = process.env.PORT || 5000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
